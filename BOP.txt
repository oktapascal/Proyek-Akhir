SELECT beban.id_beban, beban.nama_beban, detail_pembayaran_beban.jumlah_produksi, AVG(detail_pembayaran_beban.jumlah_produksi) as rata_jumlah,SUM(detail_pembayaran_beban.subtotal) as total, (AVG(total)/AVG(detail_pembayaran_beban.jumlah_produksi)) as rata FROM beban JOIN detail_beban ON beban.id_beban=detail_beban.id_beban JOIN produk ON detail_beban.id_produk=produk.id_produk JOIN detail_pembayaran_beban ON beban.id_beban=detail_pembayaran_beban.id_beban JOIN pembayaran_beban ON detail_pembayaran_beban.id_transaksi=pembayaran_beban.id_transaksi JOIN detail_pesanan ON produk.id_produk=detail_pesanan.id_produk WHERE detail_pesanan.no_pesanan="PSN-000001" AND detail_pesanan.id_produk = "PRD-000001" AND pembayaran_beban.tanggal_transaksi >= NOW()-INTERVAL 2 MONTH GROUP BY beban.id_beban
