SELECT DISTINCT nama_akun, (SELECT SUM(nominal) FROM jurnal WHERE kode_akun = '111' AND posisi_db_cr = 'Debet') debet, (SELECT SUM(nominal) FROM jurnal WHERE kode_akun = '111' AND posisi_db_cr = 'Kredit') kredit, ((SELECT SUM(nominal) FROM jurnal WHERE kode_akun = '111' AND posisi_db_cr = 'Debet') - (SELECT SUM(nominal) FROM jurnal WHERE kode_akun = '111' AND posisi_db_cr = 'Kredit')) selisih FROM jurnal JOIN akun ON akun.kode_akun=jurnal.kode_akun WHERE akun.kode_akun = '111' AND MONTH(tanggal) = '1'
